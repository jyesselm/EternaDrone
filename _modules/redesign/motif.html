		"""
		Motif object: stores 3D coordinates and 3DNA parameters of RNA Motif

		Parameters
		----------
		dir : str, the path of files required for Motif initiation: pdb,
			ref_frames.dat which is generated by 3DNA format should be 
			as follows:
				dir/dir.pdb
				dir/ref_frames.dat
			if this is not your format make sure to use the optional 
			arguments to supply this info pdb=path,ref_frames=path
		pdb : str, optional, the path of the pdb file for the 3D coordinates
			of this motif
		ref_frames: str, optional, the path of the ref_frame.dat file produced
			by 3DNA for the reference and origins of each base pair
		analysis : boolean, optional, sometimes large RNAs are loaded in as 
			Motifs this can be computationally intense. Setting analysis=False
			will greatly speed up if no assembly is to be performed on this 
			motif

		Attributes
		----------
		motif_name: str, the name of the pdb used in this motif object
		pdb : str, path of the pdb file
		ref_frames : str, path of the ref_frames file
		name : str, probably should be called dir, but its the directory
			that motif files were found (pdb,ref_frames)
		structure : ProDy structure object, holds 3D coordinate data
		hv: ProDy hiarchical object, holds 3D coordinate data
		residues: List of ProDy Residue objects
		base_pairs: List of Basepair objects, holds all the basepair info 
			determined from 3DNA
		ends: List of the Basepair objects that are at the end of Motif
			this is critical to assemble motifs together its not necessarily
			the first and last Basepairs 
		all_ends: List of BasePair objects that represent all ends, this is 
			only useful when there are more then two
		"""
		if dir != None:
			split = re.split("\/",dir)
			self.name = split[-1]
			self.dir = dir
			if pdb == None:
				pdb = dir+"/"+split[-1]+".pdb"
			if ref_frames == None:
				ref_frames = dir+"/"+"ref_frames.dat"

		elif pdb != None and ref_frames != None:
			self.pdb = pdb
			self.ref_frames = ref_frames

		else:
			raise ValueError(
				"""
				Motif was not initiated correctly!

				examples:
				m = Motif(dir="yourdir")
				where the file structure is 
				yourdir/
					yourdir.pdb
					ref_frames.dat

				m = Motif(pdb=pdb_file_path,ref_frames=ref_frames_file_path)

				""")

		self.chains = list(self.hv)
		self.real_chains = []
		self.mod_real_chains = []
		self.residues = []
		self.base_pairs = []

		allowed_res = ['A','U','G','C','rA','rU','rC','rG']

		#else:

		#determine which residue is going in the 5 prime and 3 prime

			self.assign_5_or_3_prime(e)

[docs]	def determine_chains_by_physical_connection_old(self,ignore_5prime=False):
		"""
		Initially was determing chains by residue numbering alone
		this unforunately is not sufficient as idenification of connectivity 
		is critical. Here using a 2A distance constraint between P-O3' and 
		O5' determines whether two residues are connected. Connected residues
		are stored in lists to make chains, chains are always stored 5' to 3'!

		Parameters
		----------

		Returns
		-------
		"""

		#find all chain ends, i.e. only connected to one other residue
		self.all_ends = self.find_chain_ends()

		ends = self.all_ends
		self.real_chains = []

		for e in ends:
			#alway want to go from 5' to 3' thus need to start on 5' end
			if ignore_5prime==False and not self.is_5prime_end(e):
				continue
			current = e
			seen = {}
			chain = []
			while current != None:
				chain.append(current)
				seen[current] = 1
				neighbors = self.get_neighbor_residues(current)
				next = None
				for n in neighbors:
					if not n in seen:
						next = n
				current = next
			self.real_chains.append(chain)

		if len(self.real_chains) == 0 and ignore_5prime==False:
			self.determine_chains_by_physical_connection(ignore_5prime=True)



[docs]	def find_basepair_ends(self,expected_ends=2):
		"""
		Finds which base pairs are at the end of the motif, these ends are
		used to join one motif to another through a common basepair thus
		this is absolutely critical.

		Parameters
		----------
		expected_ends : int, optional, the number of ends a motif should 
		have,  

		Returns
		-------
		"""
		expected_chain_ends = expected_ends*2

		ends = self.find_chain_ends()

		if len(ends) != expected_chain_ends:
			raise ValueError("could not find the 4 residues that are at \
				the end of this motif "+self.dir)

		#attempt to match up each residue with every other and see if they
		#are both contained in a Basepair object
		motif_ends = self.generate_bp_ends_from_chain_ends(ends)

		if len(motif_ends) != expected_ends:
			raise ValueError("could not determined the ends of this \
				motif: "+self.dir)

		return motif_ends

[docs]	def get_neighbor_residues(self,res):
		"""
		finds the residues that are connected to the current residue by 
		exhustively searching through all other residues and see if they
		can physically connect. This can be probably reworked to initalize 
		the real_chains

		Parameters
		----------
		res : ProDy residue object

		Returns
		-------
		connected_res : a list of ProDy residue objects that connected to 
			the supplied residue
		"""
		connected_res = []
		for r in self.residues:
			if res == r:
				continue 
			if self.is_connected(res,r):
				connected_res.append(r)
		return connected_res

[docs]	def parse_3dna_ref_frames_file(self,ref_frame_file):
		"""
		Parses 3DNA ref_frames.dat file to determine which residues are 
		basepaired as well as their reference frame and origin 

		Parameters
		----------
		ref_frame_file : str, path to the 3DNA ref frame file 

		Returns
		-------
		"""
		try:
			input = open(ref_frame_file)
			lines = input.readlines()
			lines.pop(0)
		except IOError:
			print "3DNA ref frame file="+ref_frame_file+" does not exists "+\
				"please check the path\n" 
			raise

		base_pairs = []

		p = re.compile('(\w+)\:\.*(\d+)\_\:\[\.*\w+\]\w+ \- (\w+)\:\.*(\d+)')
		start_bp = 0
		r = []
		d = []
		bps = None
		for l in lines:
			if l[0:3] == "...":
				start_bp = 1
				if p.search(l):
					m = p.search(l)
					bps = m.groups()
				continue

			if start_bp == 0:
				continue
			if start_bp == 1:
				split = re.split("\s+", l)
				d = [ float(x) for x in split[1:4] ]
				start_bp += 1
			elif start_bp > 1:
				split = re.split("\s+", l)
				r.append([ float(x) for x in split[1:4] ])
				start_bp += 1
			#have everything, generate BasePair object
			if start_bp == 5:
				res1 = self.find_residue(bps[0],float(bps[1])) 
				res2 = self.find_residue(bps[2],float(bps[3]))
				bp = Basepair(np.array(d),np.array(r),res1,res2)
				base_pairs.append(bp)
				r = []
				d = []

		if len(base_pairs) == 0:
			raise ValueError("there are not base pairs declared in this ref \
				file="+ref_frame_file)

		self.base_pairs = base_pairs

[docs]	def find_base_pair(self,res1,res2):
		"""
		locates a Basepair object by the residue num and chain id of both 
		residues in the base pair. Order of res1 and res2 is not inforced

		Parameters
		----------
		res1: array, contains the residue num and chain id of first residue
		res2: array, contains the residue num and chain id of second residue

		Returns
		-------
		returns a Basepair object
		"""
		for bp in self.base_pairs:
			if bp.res1.getResnum() == res1[1] and bp.res2.getResnum() == res2[1] \
			   and bp.res1.getChid() == res1[0] and bp.res2.getChid() == res2[0]:
				return bp
			if bp.res1.getResnum() == res2[1] and bp.res2.getResnum() == res1[1] \
			   and bp.res1.getChid() == res2[0] and bp.res2.getChid() == res1[0]:
				return bp

		return None 


	def find_base_pair_for_res(self,res):
		for bp in self.base_pairs:
			if bp.res1.getResnum() == res.getResnum() and bp.res1.getChid() == res.getChid() and bp.res1.getResname() == res.getResname():
				return bp
			if bp.res2.getResnum() == res.getResnum() and bp.res2.getChid() == res.getChid() and bp.res2.getResname() == res.getResname():
				return bp
		return None


[docs]	def generate_motifchain_constraints(self,sbp_string,ebp_string,
										remove_inbetween_residues=False):
		"""
		Parses the command line arguments and finds both the start and end 
		basepair objects required as input for a MotifChain object. 

		Parameters
		----------
		sbp_string: str, description of the start basepair in the format
		ChainNameResidueNum-ChainNameResidueNum, ex: A100-B32, which would
		be a basepair between residue A100 and B32 where A and B are their 
		chain names and 100 and 32 are the residue numbers 
		bp_2_string: str, description of the end basepair see sbp_string for 
		more info
		remove_inbetween_residues, optional, bool removes residues inbetween,
		the two basepairs to avoid steric collisions this is for redesign of 
		chains

		Returns
		-------
		start_bp: BasePair object, hold information about the start basepair,
		this is where the motif chain will build from
		end_bp: BasePair object, this is the where the algorithm is attempting
		to build to
		centers: Array of 3D points (x,y,z), represents steric centers of 
		residues that will remain fixed and should not be collided with, these
		will be check doing building to void steric clashes.
		"""

		bp_1 = re.split("\-",sbp_string)
		bp_2 = re.split("\-",ebp_string) 

		res1 = [bp_1[0][:1], float(bp_1[0][1:])]
		res2 = [bp_1[1][:1], float(bp_1[1][1:])]

		start_bp = self.find_base_pair(res1,res2)

		if start_bp == None:
			raise ValueError("Cannot find base pair "+bp_1_string)

		res1 = [bp_2[0][:1], float(bp_2[0][1:])]
		res2 = [bp_2[1][:1], float(bp_2[1][1:])]

		end_bp = self.find_base_pair(res1,res2)

		if end_bp == None:
			raise ValueError("Cannot find base pair "+bp_2_string)

		if remove_inbetween_residues == True:
			residues = self.remove_residues_between(start_bp,end_bp)
		else:
			residues = None

		centers = self.get_residue_steric_bead_centers(exclude_bps=[end_bp],
													   exclude_r=residues)
		self.assign_5_or_3_prime(start_bp)
		self.assign_5_or_3_prime(end_bp)

		end_bp.calc_sugar_centers()
		start_bp.calc_sugar_centers()

		return start_bp,end_bp,centers



[docs]	def remove_residues_between(self,start_bp,end_bp,exit=True):
		"""
		Removes residues between the start and end base pairs, this is done 
		to remove possible steric clashes while building a new motif chain

		Parameters
		----------
		start_bp, Basepair object, the start of the new motif chain
		end_bp, Basepair object, the end of the new motif chain

		Returns
		-------
		returns a list of residues that will be excluded from 

		"""

		#Figure out which residue is likely on the same chain as the other
		#by comparing residue number, might be good idea to also check
		#chain name, residues that are closest in number are grouped into "pairs"
		pairs = []
		diff1 = start_bp.res1.getResnum() - end_bp.res1.getResnum()
		diff2 = start_bp.res1.getResnum() - end_bp.res2.getResnum()

		if abs(diff1) > abs(diff2):
			pairs.append([start_bp.res1,end_bp.res2])
			pairs.append([start_bp.res2,end_bp.res1])
		else:
			pairs.append([start_bp.res2,end_bp.res2])
			pairs.append([start_bp.res1,end_bp.res1])

		mod_real_chains = []
		exclude_from_beads = []
		used_chains = []
		saved_mod_chains = []

		#keeps track of what residues have already been seen its very easy to 
		#visit a residue multiple times and end up including it by accident
		seen_res = {}
		for p in pairs:
			#print p
			start_pos = -1
			end_pos = -1
			start_chain = None
			end_chain = None

			#check each residue is on the same chain then finding the position
			#of both in the array will allow removal of all the residues between
			#by a simple array splice operation
			for c in self.real_chains:
				for i,r in enumerate(c):
					if r == p[0]:
						start_pos = i
						start_chain = c
					elif r == p[1]:
						end_pos = i
						end_chain = c

			if start_chain == None and exit:
				raise ValueError("Could not find what chain "+str(p[0])+" belongs to")
			elif start_chain == None:
				return None

			if start_chain != end_chain and exit:
				raise ValueError("Cannot remove residues inbetween if on two different chains")
			elif start_chain != end_chain:
				return None

			#array splicing is always in the form [A:B] where A < B, check to 
			#make sure thats true
			if start_pos > end_pos:
				temp = start_pos
				start_pos = end_pos
				end_pos = temp

			removed_portion = start_chain[start_pos+1:end_pos]
			for r in removed_portion:
				#print r
				exclude_from_beads.append(r)
				seen_res[r] = 1

			saved_mod_chains.append(start_chain[:start_pos+1])
			saved_mod_chains.append(start_chain[end_pos:])

			used = 0
			for uc in used_chains:
				if start_chain == uc:
					used = 1
					break

			if not used:
				used_chains.append(start_chain)

		saved_chain = []
		for c in saved_mod_chains:
			for r in c:
				if r in seen_res:
					if len(saved_chain) > 0:
						mod_real_chains.append(saved_chain)
						saved_chain = []
				else:
					seen_res[r] = 1
					saved_chain.append(r)
					
			if len(saved_chain) > 0:
				mod_real_chains.append(saved_chain) 
				saved_chain = []

		for c in self.real_chains:
			used = 0
			for uc in used_chains:
				if c == uc:
					used = 1
					break
			if not used:
				mod_real_chains.append(c)

		self.mod_real_chains = mod_real_chains

		return exclude_from_beads
