	"""
	sets up a directory to be loaded in as a motif object. Motifs are loaded
	in as a 

	"""

	split = re.split("\/",pdb_path)
	name = split[-1]
	name_no_exd = name[:-4]
	if not os.path.isdir(name_no_exd): 
		os.mkdir(name_no_exd)
	elif os.path.isfile(name_no_exd+"/ref_frames.dat"):
		os.remove(name_no_exd+"/ref_frames.dat")
	shutil.copy(pdb_path,name_no_exd+"/"+name)
	os.chdir(name_no_exd)
	subprocess.call("find_pair "+name+" 2> /dev/null stdout | analyze stdin >& /dev/null", shell=True)
	os.chdir("..")
	return name_no_exd 


def unitarize(R):
    """
    Enforce unitarity of the input matrix using Gram-Schmidt process.
    This function overwrites the input matrix.

    Parameters
    ----------
    R : ndarray, shape (N,3,3)
        (List of) input matrix to be unitarized.

    Returns
    -------
    R : ndarray, shape (N,3,3)
        Unitarized input matrix.
    """
    if len(R.shape) == 2:  # 1D case
        R[0] /= math.sqrt(R[0].dot(R[0]))
        R[1] -= R[1].dot(R[0]) * R[0]
        R[1] /= math.sqrt(R[1].dot(R[1]))
        R[2] -= R[2].dot(R[0]) * R[0]
        R[2] -= R[2].dot(R[1]) * R[1]
        R[2] /= math.sqrt(R[2].dot(R[2]))
    else:  # 2D case
        R[:, 0] /= np.sqrt(np.sum(R[:, 0] ** 2, axis=1))[:, np.newaxis]
        R[:, 1] -= (
            np.einsum('ij,ij->i', R[:, 1], R[:, 0])[:, np.newaxis] * R[:, 0])
        R[:, 1] /= np.sqrt(np.sum(R[:, 1] ** 2, axis=1))[:, np.newaxis]
        R[:, 2] -= (
            np.einsum('ij,ij->i', R[:, 2], R[:, 0])[:, np.newaxis] * R[:, 0])
        R[:, 2] -= (
            np.einsum('ij,ij->i', R[:, 2], R[:, 1])[:, np.newaxis] * R[:, 1])
        R[:, 2] /= np.sqrt(np.sum(R[:, 2] ** 2, axis=1))[:, np.newaxis]
    return R

def align_motif_end_to_basepair(ref_bp,motif,end_pos,previous_transform=None):
	"""
	This is the workhorse of the entire suite. Aligns one end of a motif to
	the reference frame and origin of a Basepair object.

	Parameters
	----------
	ref_bp: Basepair object, this is the reference base pair that the end base
		pair of the motif will align to
	motif: Motif object, the motif that needs to be aligned to the reference
		base pair
	end_pos: int, the end position in motif that will be aligned to ref_bp. 
		This corresponds to the position in the array of ends stored in each 
		Motif object. Most motifs have two ends thus this value is 0 or 1
	previous_transform: ProDy transformation object, optional, represents 
		the transform that orientated ref_bp to its current orientation. 
		This is required for a sequence of transformations

	Returns
	-------
	Returns a ProDy transformation object that represents the transformation
	required to align the current motif to the ref_bp. This does not need to be
	applied to the current motif but is required for additional motifs to be 
	aligned to the current motif. 
	"""

	#rotate new motif so its end base pair is oriented in the same ref frame
	#as start base pairs
	r1 = ref_bp.r
	r2 = motif.ends[end_pos].r
	r =  unitarize(r1).T.dot(unitarize(r2))
	rot_origin = calcCenter(motif.ends[end_pos].atoms)
	t = Transformation(r,-rot_origin)
	applyTransformation(t,motif.structure)

	#Apply previous transforms to orient to the current position in the motif
	#chain
	if(previous_transform != None):
		applyTransformation(previous_transform, motif.structure)

	bp_pos_diff = calcCenter(ref_bp.atoms) - \
				  calcCenter(motif.ends[end_pos].atoms)

	t2 = Transformation(np.eye(3), bp_pos_diff)

	applyTransformation(t2,motif.structure)

	#Since alignment is by the center of the base pair often the alignment 
	#is a bit off a needs to be offset a small amount. This is done by aligning
	#the C1' of both sugars for the overlapping basepair
	##########################################################################

	lm_res1 = ref_bp.res1
	lm_res2 = ref_bp.res2

	nm_res1 = motif.ends[end_pos].res1
	nm_res2 = motif.ends[end_pos].res2

	diff_1 = calcCenter(lm_res1) - calcCenter(nm_res1)
	diff_2 = calcCenter(lm_res1) - calcCenter(nm_res2)

	sugar_11 = lm_res1.getAtom("C1'")
	sugar_12 = lm_res2.getAtom("C1'")

	if np.linalg.norm(diff_1) < np.linalg.norm(diff_2):
		sugar_21 = nm_res1.getAtom("C1'")
		sugar_22 = nm_res2.getAtom("C1'")
	else:
		sugar_21 = nm_res2.getAtom("C1'")
		sugar_22 = nm_res1.getAtom("C1'")


	sugar_diff_1 = sugar_11.getCoords() - sugar_21.getCoords()
	sugar_diff_2 = sugar_12.getCoords() - sugar_22.getCoords()

	moveAtoms(motif.structure, by=(sugar_diff_1+sugar_diff_2)/2)

	##########################################################################

	#update previous transform so the next motif being added can be aware of 
	#the orientation change performed by adding the current motif
	if previous_transform != None:
		final_t = Transformation(previous_transform._matrix.dot(t._matrix))
	else:
		final_t = t

	return final_t

def align_sugars(ref_bp,new_bp):
	lm_res1 = ref_bp.res1
	lm_res2 = ref_bp.res2

	nm_res1 = new_bp.res1
	nm_res2 = new_bp.res2

	diff_1 = calcCenter(lm_res1) - calcCenter(nm_res1)
	diff_2 = calcCenter(lm_res1) - calcCenter(nm_res2)

	sugar_11 = lm_res1.getAtom("C1'")
	sugar_12 = lm_res2.getAtom("C1'")

	if np.linalg.norm(diff_1) < np.linalg.norm(diff_2):
		sugar_21 = nm_res1.getAtom("C1'")
		sugar_22 = nm_res2.getAtom("C1'")
	else:
		sugar_21 = nm_res2.getAtom("C1'")
		sugar_22 = nm_res1.getAtom("C1'")


	sugar_diff_1 = sugar_11.getCoords() - sugar_21.getCoords()
	sugar_diff_2 = sugar_12.getCoords() - sugar_22.getCoords()

	moveAtoms(new_bp.atoms, by=(sugar_diff_1+sugar_diff_2)/2)
	return (sugar_diff_1+sugar_diff_2)/2

def get_sequence_and_ss_structure(motif):

	#if chains have not been determined yet run analysis
	if len(motif.real_chains) == 0:
		motif.determine_chains_by_physical_connection()

	if len(motif.real_chains) > 1:
		raise ValueError("cannot generate secondary for motif with then one chain")

	renumbered = {}
	mc_res = []
	for chain in motif.real_chains:
		rnum = 1
		for r in chain:
			renumbered[r.getResnum()] = rnum
			if r.getResnum() < 100:
				mc_res.append(rnum-1)
			rnum += 1
		break

	f = open(motif.dir+"/"+motif.name+".out")
	lines = f.readlines()
	f.close()

	p = re.compile("(\d+)\_\:\[\.+\w\]\w([\-\+\*]+)\w\[\.+\w]\:\.*(\d+)")

	bp_pattern = {}
	for l in lines:
		if p.search(l):
			m = p.search(l)
			groups =  m.groups()
			wc = 1
			line_count = 0
			for s in groups[1]:
				if s == "-":
					line_count += 1

			if line_count < 4:
				wc = 0

			bp_pattern[groups[0]+"-"+groups[2]] = wc
			#if abs(int(groups[0])-int(groups[2])) == 1:
				#print groups[0],groups[2]

	bps = motif.base_pairs

	bps.sort(key = lambda x: renumbered[x.res1.getResnum()])

	length = rnum-1

	sequence = ""
	ss_structure = ["." for x in range(0,length)]
	design_sequence = ["N" for x in range(0,length)]

	bp_pairs = {}

	for r in motif.real_chains[0]:
		sequence += r.getResname()

	for bp in bps:

		res1_pos = bp.res1.getResnum()
		res2_pos = bp.res2.getResnum()

		##print res1_pos,res2_pos

		bpstr = bp.res1.getResname()+bp.res2.getResname()

		bpp = 0
		is_bp = 0
		reverse = 0
		if bpstr == 'GC' or bpstr == 'CG' or bpstr == 'AU' or bpstr == 'UA': 
		#or bpstr == 'GU' or bpstr == 'UG': 
			bp_name = str(bp.res1.getResnum())+"-"+str(bp.res2.getResnum())

			if bp_name in bp_pattern:
				bpp = bp_pattern[bp_name]
			else:
				bp_name = str(bp.res2.getResnum())+"-"+str(bp.res1.getResnum())
				bpp = bp_pattern[bp_name]
				reverse = 1

			if res1_pos not in renumbered or res2_pos not in renumbered:
				return [[],""]

			if bpp == 1:
				if reverse == 0:
					res1_pos = renumbered[res1_pos]-1
					res2_pos = renumbered[res2_pos]-1
				else:
					res1_pos = renumbered[res2_pos]-1
					res2_pos = renumbered[res1_pos]-1

				if res1_pos < res2_pos:
					ss_structure[res1_pos]="("
					ss_structure[res2_pos]=")"
				else:
					ss_structure[res1_pos]=")"
					ss_structure[res2_pos]="("
				is_bp =1	
				bp_pairs[res1_pos] = res2_pos
				bp_pairs[res2_pos] = res1_pos
	

	##print sequence
	##print "".join(ss_structure)

	for i,e in enumerate(sequence):
		design_sequence[i] = e

	for r_num in mc_res:
		#print r_num
		if ss_structure[r_num] == ".":
			continue
		if rnum-1 > -1 and ss_structure[r_num-1] == ".":
			continue
		if rnum+1 < len(sequence) and ss_structure[r_num+1] == ".":
			continue
		pair_num = bp_pairs[r_num]
		if pair_num-1 > -1 and ss_structure[pair_num-1] == ".":
			continue
		if pair_num+1 < len(sequence) and ss_structure[pair_num+1] == ".":
			continue
		design_sequence[r_num] = "N" 


	return ss_structure,sequence,design_sequence

def distance(p1,p2):
	return math.sqrt((p2[0] - p1[0]) ** 2 +
                     (p2[1] - p1[1]) ** 2 +
                     (p2[2] - p1[2]) ** 2)   

def distance_squared(p1,p2):
	return ((p2[0] - p1[0]) ** 2 +
            (p2[1] - p1[1]) ** 2 +
            (p2[2] - p1[2]) ** 2)

def merge_chains(base_chain,new_chain,overlap_threshold=5,remove_overlap=True):
		base_chain_end_centers = [[calcCenter(base_chain[0]),0],
								  [calcCenter(base_chain[-1]),1]]

		new_chain_end_centers = [[calcCenter(new_chain[0]),0],
								 [calcCenter(new_chain[-1]),1]]

		best = 10000
		pattern = []

		for bc_end in base_chain_end_centers:
			for nc_end in new_chain_end_centers:
				d = distance(bc_end[0],nc_end[0])
				if d < best:
					best = d
					pattern = [bc_end[1],nc_end[1]]

		if best < overlap_threshold:
			merged = []
			if pattern[0] == 0:
				base_chain = base_chain[::-1]
			if pattern[1] == 1:
				new_chain = new_chain[::-1]
			#new_chain.pop(0)
			for c in base_chain:
				merged.append(c)
			start = 0
			if remove_overlap:
				start = 1
			for i in range(start,len(new_chain)):
				merged.append(new_chain[i])
			return merged
		else:
			return None

def write_chain_to_pdb(name,chain):
	atoms = None
	for i,r in enumerate(chain):
		if i == 0:
			atoms = r.copy()
		else:
			atoms += r.copy()
	writePDB(name,atoms)

	f = open(name)
	lines = f.readlines()
	f.close()

	lines.pop(0)
	f = open(name,"w")
	for l in lines:
		f.write(l)
	f.close()

def is_connected_one_way(res1,res2):
	"""
	Determines whether res1 and res2 are connected, order does matter,
	checks to see if P atoms of res1 is connected to either the O3 or O5
	atom of res2

	Parameters
	----------
	res1 : ProDy Residue Object, first residue 

	res2 : ProDy Residue Object, second residue

	Returns
	-------
	return 1 if res1 and res2 are connected, return 0 if not
	"""

	if res1 == None or res2 == None:
		return 0

	p_atom = res1.getAtom("P")
	o3_atom = res2.getAtom("O3'")
	o5_atom = res2.getAtom("O5'")

	#for some motifs O3' was renamed O2', check to make sure there isnt
	#an O2' atom
	if o3_atom == None:
		o3_atom = res2.getAtom("O2'")

	if p_atom == None or o3_atom == None or o5_atom == None:
		return 0

	if calcDistance(p_atom,o3_atom) < 2 or calcDistance(p_atom,o5_atom) < 2:
		return 1
	else:
		return 0

def is_connected(res1,res2):
	"""
	Determines whether res1 and res2 are connected, order does not matter
	checks using distance of 2A from P-O3 or P-O5

	Parameters
	----------
	res1 : ProDy Residue Object, first residue 

	res2 : ProDy Residue Object, second residue

	Returns
	-------
	return 1 if res1 and res2 are connected, return 0 if not
	"""
	
	if is_connected_one_way(res1,res2):
		return 1
	elif is_connected_one_way(res2,res1):
		return 1
	else:
		return 0
